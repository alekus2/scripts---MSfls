
def calcular_maximo(projeto, talhao):
    valores = 0
    for row in arcpy.da.SearchCursor("Shape_IFQ12_teste", ["ID_PROJETO", "ID_TALHAO"]):
        if row[0] == projeto and row[1] == talhao:
            return valores + 1
        else: #desejo contar a quantidade de vezes que a concatenação de id projeto e id talhao se repetem para formar o numero de parcelas total do talhao
            return valores - 1
    contar_parcelas = len(valores)  
    return contar_parcelas
