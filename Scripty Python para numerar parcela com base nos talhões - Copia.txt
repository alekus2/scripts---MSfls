def calcular_maximo(talhao):
    max_parcelas = {}
    valores = [int(row[1]) for row in arcpy.da.SearchCursor("Nome da sua tabela", ["Index", "nm_parcela"]) if row[0] == talhao]
    max_parcelas[talhao] = max(valores)
    return max_parcelas[talhao]

def matizador(area):
    # Utilize um dicionário para armazenar os talhões e suas parcelas
    talhoes_parcelas = {}
    
    # Obtenha os valores da área da tabela "piracicaba_talhão"
    valores = [row[0] for row in arcpy.da.SearchCursor("piracicaba_talhão", ["AREA_HA"])]
    
    # Calcule a quantidade de parcelas, que é a soma das áreas dividida por 5
    total_area = sum(valores)
    talhoes_parcelas[area] = total_area / 5
    
    return talhoes_parcelas[area]

# Chame a função passando o valor da área atual
