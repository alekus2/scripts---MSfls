def calcular_maximo(projeto, talhao, nm_parcela):
    # Cria uma lista para armazenar os valores das parcelas
    valores = []

    # Percorre as características da camada atual para contar as parcelas
    for row in arcpy.da.SearchCursor("sua_layer_aqui", ["ID_PROJETO", "ID_TALHAO", "nm_parcela"]):
        if row[0] == projeto and row[1] == talhao and row[2] == nm_parcela:
            valores.append(row)

    # Calcula o máximo das parcelas encontradas
    max_parcelas = len(valores)  # Conta quantas parcelas existem para a combinação

    return max_parcelas
