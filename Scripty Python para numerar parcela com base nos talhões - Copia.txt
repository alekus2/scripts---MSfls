def calcular_maximo(talhao):
    max_parcelas = {}
    valores = [int(row[1]) for row in arcpy.da.SearchCursor("Nome da sua tabela", ["Index", "nm_parcela"]) if row[0] == talhao]
    max_parcelas[talhao] = max(valores)
    return max_parcelas[talhao]
